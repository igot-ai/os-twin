using Game.{Feature}.Config;
using Game.{Feature}.Logic;
using Game.{Feature}.Models;
using VContainer;

namespace Game.{Feature}
{
    /// <summary>
    /// VContainer installer for {Feature} services.
    /// Usage: new {Feature}Installer(_config).Install(builder);
    /// </summary>
    public class {Feature}Installer : IInstaller
    {
        private readonly {Feature}Config _config;

        public {Feature}Installer({Feature}Config config)
        {
            _config = config;
        }

        public void Install(IContainerBuilder builder)
        {
            // Register config as singleton instance
            builder.RegisterInstance(_config).AsSelf();

            // Register model (scoped to gameplay lifetime)
            builder.Register<{Feature}Model>(Lifetime.Scoped).AsSelf();

            // TODO: Register additional services
            // builder.Register<{Feature}Service>(Lifetime.Scoped).As<I{Feature}Service>();
        }
    }
}
