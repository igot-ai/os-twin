using Cysharp.Threading.Tasks;
using Sun.Core.UI;
using UniRx;
using UnityEngine;
using UnityEngine.UI;
using Game.{Feature}.UI.Models;

namespace Game.{Feature}.UI.Views
{
    /// <summary>
    /// UI screen for {Feature}. Extends MainView for UIManager integration.
    /// </summary>
    public class {Feature}UI : MainView<{Feature}UIModel>
    {
        [Header("View References")]
        [SerializeField] private CanvasGroup _canvasGroup;
        // [SerializeField] private Button _btnAction;

        protected override void OnSetModel({Feature}UIModel viewModel)
        {
            // Bind reactive properties to UI elements
            viewModel.IsInteractable
                .Subscribe(isInteractable => _canvasGroup.interactable = isInteractable)
                .AddTo(compositeDisposable);

            // Bind button clicks
            // _btnAction.OnClickAsObservable()
            //     .Subscribe(_ => viewModel.OnAction?.Invoke())
            //     .AddTo(compositeDisposable);
        }

        public override async UniTask OnViewStart()
        {
            await base.OnViewStart();
            uiModel.IsInteractable.Value = true;
        }

        public override async UniTask OnUIRemoved()
        {
            await base.OnUIRemoved();
        }
    }
}
