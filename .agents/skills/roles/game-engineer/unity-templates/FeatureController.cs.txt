using System.Threading;
using Cysharp.Threading.Tasks;
using Game.{Feature}.Logic;
using Game.{Feature}.Models;
using Game.{Feature}.Views;
using Game.{Feature}.Config;
using Game.{Feature}.Enums;
using Sun.Audio;
using Sun.Haptic;
using UniRx;
using UnityEngine;
using VContainer;

namespace Game.{Feature}
{
    /// <summary>
    /// Orchestrates {Feature} game flow: logic, views, and side effects.
    /// </summary>
    public partial class {Feature}Controller : MonoBehaviour
    {
        [SerializeField] private {Feature}View _{feature}View;

        // Reactive state exposed to presenters/UI
        public ReactiveProperty<E{Feature}State> State { get; } = new(E{Feature}State.None);
        public ISubject<Unit> OnStart { get; } = new Subject<Unit>();
        public ISubject<bool> OnFinish { get; } = new Subject<bool>();

        // Injected dependencies
        private {Feature}Model _model;
        private {Feature}Config _config;
        private IAudioService _audioService;
        private IHapticService _hapticService;

        // Game logic (pure C#)
        public {Feature}GameLogic GameLogic { get; private set; }

        private CompositeDisposable _disposables = new();
        private bool _isServiceInjected;

        [Inject]
        public void Construct(
            {Feature}Model model,
            {Feature}Config config,
            IAudioService audioService,
            IHapticService hapticService)
        {
            _model = model;
            _config = config;
            _audioService = audioService;
            _hapticService = hapticService;
            _isServiceInjected = true;
        }

        private void Start()
        {
            Initialize().Forget();
        }

        private async UniTask Initialize()
        {
            var ct = this.GetCancellationTokenOnDestroy();
            await UniTask.WaitUntil(() => _isServiceInjected, cancellationToken: ct);

            GameLogic = new {Feature}GameLogic();
            await _{feature}View.Initialize();

            _disposables.Clear();
            SetupEventSubscriptions();
        }

        private void SetupEventSubscriptions()
        {
            State.Subscribe(OnStateChanged).AddTo(_disposables);
            GameLogic.Events.OnFinished.Subscribe(isWin => HandleFinished(isWin).Forget()).AddTo(_disposables);
        }

        private void OnStateChanged(E{Feature}State state)
        {
            // TODO: Handle state transitions (enable/disable input, etc.)
        }

        private async UniTask HandleFinished(bool isWin)
        {
            if (State.Value == E{Feature}State.Victory || State.Value == E{Feature}State.GameOver)
                return;

            State.Value = isWin ? E{Feature}State.Victory : E{Feature}State.GameOver;
            OnFinish.OnNext(isWin);
            await UniTask.CompletedTask;
        }

        /// <summary>
        /// Start or restart the feature gameplay.
        /// </summary>
        public async UniTask StartGame(CancellationToken ct = default)
        {
            if (ct == default) ct = this.GetCancellationTokenOnDestroy();

            State.Value = E{Feature}State.None;
            GameLogic.Setup();
            // TODO: Load data, setup view
            State.Value = E{Feature}State.Idle;
            OnStart.OnNext(Unit.Default);
            await UniTask.CompletedTask;
        }

        private void Update()
        {
            if (State.Value == E{Feature}State.Playing)
            {
                GameLogic.Tick(Time.deltaTime);
            }
        }

        private void OnDestroy()
        {
            _disposables.Dispose();
        }
    }
}
