using System.Threading;
using Cysharp.Threading.Tasks;
using Game.{Feature}.Enums;
using UnityEngine;

namespace Game.{Feature}
{
    public partial class {Feature}Controller
    {
        private CancellationTokenSource _debugCts;

        /// <summary>
        /// Force win the current game.
        /// </summary>
        public void DebugWin()
        {
            if (GameLogic == null)
            {
                Debug.LogWarning("Cannot win: Game not initialized");
                return;
            }

            Debug.Log("[Debug] Force winning");
            GameLogic.Events.OnFinished.OnNext(true);
        }

        /// <summary>
        /// Force lose the current game.
        /// </summary>
        public void DebugLose()
        {
            if (GameLogic == null)
            {
                Debug.LogWarning("Cannot lose: Game not initialized");
                return;
            }

            Debug.Log("[Debug] Force losing");
            GameLogic.Events.OnFinished.OnNext(false);
        }

        /// <summary>
        /// Log current game state for debugging.
        /// </summary>
        public void DebugLogState()
        {
            Debug.Log($"[Debug] State: {State.Value}, Logic State: {GameLogic?.State.Value}");
        }
    }
}
