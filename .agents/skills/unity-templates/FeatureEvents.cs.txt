using UniRx;

namespace Game.{Feature}.Logic
{
    // Event structs — readonly, zero allocation
    public readonly struct {Feature}ActionEvent
    {
        public readonly int TargetId;
        public {Feature}ActionEvent(int targetId) { TargetId = targetId; }
    }

    public readonly struct {Feature}CompletedEvent
    {
        public readonly bool IsSuccess;
        public {Feature}CompletedEvent(bool isSuccess) { IsSuccess = isSuccess; }
    }

    /// <summary>
    /// Event publishers for {Feature} game logic.
    /// </summary>
    public class {Feature}Events
    {
        public ISubject<{Feature}ActionEvent> OnActionPerformed { get; } = new Subject<{Feature}ActionEvent>();
        public ISubject<{Feature}CompletedEvent> OnCompleted { get; } = new Subject<{Feature}CompletedEvent>();
        public ISubject<bool> OnFinished { get; } = new Subject<bool>();
    }
}
